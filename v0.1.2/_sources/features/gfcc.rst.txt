  import scipy.io.wavfile
  import spafe.utils.vis as vis
  from spafe.features.mfcc import gfcc


  #read wave file

  # compute gfccs

  # visualize features
  vis.visualize(gfccs, LMFCC Coefficient Index','Frame Index')
