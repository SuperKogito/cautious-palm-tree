  import scipy.io.wavfile
  import spafe.utils.vis as vis
  from spafe.features.mfcc import lfcc


  #read wave file

  # compute lfccs

  # visualize features
  vis.visualize(lfccs, LMFCC Coefficient Index','Frame Index')
